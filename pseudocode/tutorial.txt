Tutorial Pseudocod
BEGIN tutorialClass()
  Font1 = new font
  Font2 = new font
  Font3 = new font
  
  Stage = 1
  Stages = {
    {
      title = "Moving Around"
      text = "Used the 'A' and 'D' keys to move left and right across the screen."
      BEGIN Test()
        RETURN Walk Time > 1
      END Test
    },
    {
      title = "Shooting"
      text = "Shooting is a fundamental part of the game, move the mouse around and click to shoot"
      BEGIN test()
        RETURN IF mouse clicked
      END test
    }
    {
      title = "Reloading"
      text = "After using up bullets, try reloading by pressing 'r'. The gun is also automatically reloaded when you run out of bullets in the clip."
      BEGIN test()
        RETURN ('r' is pressed AND gun ammo is less than in the ammo clip)
      END test
    }
    {
      title = "Zombies"
      text = "Zombies will appear from either the left or right side and will attempt to get to you, try shooting a zombie until it dies."
      BEGIN test()
        RETURN (gun kills = 1)
      END test
      BEGIN setup()
        Ak47.locked = False
        INSERT zombieClass() INTO zombieList
        First zombie attack damage = 0
      END setup
    }
    {
      title = "Gun Crate"
      text = "Collecting gun crates gives you the gun for the rest of the wave. Collect the AK47 crate on the ground and equip it with the number keys."
      BEGIN test()
        RETURN selected gun from inventory = ak47
      END test
    }
    {
      title = "Tutorial Complete!"
      text = "All done, click the button in the top left to head back to the main menu and start a game with the new game button."
      BEGIN test()
        RETURN False
      END test
    }
  }
  
  BEGIN render(dt)
    Display background image
    
    FOR EACH zombie IN zombieList DO
      Display zombie
    END 
    
    Display player image
    
    FOR EACH item IN droppedItems DO
      Display item
    END 
    
    Display inventory
    
    Display main menu button
    IF mouseInRect(main menu button) THEN
      Display transparent colour over button
    ENDIF
    
    Display tutorial box
    
  BEGIN update()
    IF mouse clicked AND mouseInRect(main menu button) THEN
      player.health.cur = 0
      main.state = "main"
      mouseClicked = false

    ELSE
      IF stages[stage].test() THEN
        INCREMENT stage
      ENDIF

      FOR EACH zombie IN zombieList DO
        zombie.update(dt)
      END

      FOR EACH zombie IN zombieList DO
        IF zombie.deathTime >= zombie.deathTurningTime THEN
          REMOVE zombie FROM zombieList
        ENDIF
      END

      inv.update()

      player.update(dt)

    ENDIF
  END update
END tutorialClass
